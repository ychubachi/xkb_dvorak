1) Move xkb_dvorak/ to ~/.xkb/
  mv xkb_dvorak ~/.xkb

2) Add following commands to .bashrc
  xkbcomp -I$HOME/.xkb ~/.xkb/keymap/mykbd $DISPLAY 2> /dev/null

4) Clear cache if necessary
  sudo rm /var/lib/xkb/*.xkm

3) Restart
